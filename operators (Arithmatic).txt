class Arithmatic_operations
{
	static void Main()
	{
		int x = 10;
		int y = 8;
		int add,sub,multi,div_int,modulus;
		float div_float;
		add = x+y;
		console.writeline("{0}+{1}={2}",x,y,add);

		console.writeline("{0}-{1}={2}",x,y,add);

		modulus = x%y;

	}
}
